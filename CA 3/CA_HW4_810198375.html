import numpy as np
import random 
import matplotlib.pyplot as plt  
N = 100
  
circle_points= 0
square_points= 0
xc = []
yc = []
xs = []
ys = []
# Total Random numbers generated= possible x 
# values* possible y values 
for i in range(N): 
  
    # Randomly generated x and y values from a 
    # uniform distribution 
    # Rannge of x and y values is -1 to 1 
    rand_x= random.uniform(-1, 1) 
    rand_y= random.uniform(-1, 1) 
  
    # Distance between (x, y) from the origin 
    origin_dist= rand_x**2 + rand_y**2
  
    # Checking if (x, y) lies inside the circle 
    if origin_dist<= 1: 
        circle_points+= 1
        xc.append(rand_x)
        yc.append(rand_y)
    else :
        xs.append(rand_x)
        ys.append(rand_y)

    square_points+= 1
  
    # Estimating value of pi, 
    # pi= 4*(no. of points generated inside the  
    # circle)/ (no. of points generated inside the square) 
    pi = 4* circle_points/ square_points 
plt.scatter(xc, yc, marker='.', color='g')
plt.scatter(xs, ys, marker='.', color='r' )
print("Final Estimation of Pi=", "%.22f" % pi)
pi_accuracy = abs(np.pi - pi)/2
print("accuracy for  evaluating Pi=", "%.22f" % pi_accuracy) 

Final Estimation of Pi= 3.0800000000000000710543
accuracy for  evaluating Pi= 0.0307963267948965224718
